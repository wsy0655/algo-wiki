    йҖҶе…ғ

    ж•°еӯҰеҪ’зәіжі•еҲқжӯҘ¶

    ж•°еӯҰеҪ’зәіжі•жҳҜдёҖз§ҚиҜҒжҳҺж–№жі•пјҢйҖҡеёёиў«з”ЁдәҺиҜҒжҳҺжҹҗдёӘз»ҷе®ҡе‘ҪйўҳеңЁж•ҙдёӘпјҲжҲ–иҖ…еұҖйғЁпјүиҮӘз„¶ж•°иҢғеӣҙеҶ…жҲҗз«ӢгҖӮ

    ж ёеҝғжҖқжғіпјҡ еҰӮжһңжҹҗд»¶дәӢеҜ№з¬¬дёҖдёӘдәәжҲҗз«ӢпјҢе№¶дё”еҜ№з¬¬ $k$ дёӘдәәжҲҗз«Ӣе°ұиғҪжҺЁеҮәеҜ№з¬¬ $k+1$ дёӘдәәд№ҹжҲҗз«ӢпјҢйӮЈд№Ҳе®ғеҜ№жүҖжңүдәәйғҪжҲҗз«ӢгҖӮ

    йҖҡдҝ—зұ»жҜ”пјҡ жҺ’йҳҹжҺЁеӨҡзұіиҜәйӘЁзүҢпјҡ

    • дёҖдёӘйӘЁзүҢеҖ’дёӢдәҶпјҲйӘҢиҜҒиө·е§Ӣпјү
    • жҜҸдёӘеҖ’дёӢзҡ„йӘЁзүҢйғҪдјҡжҺЁеҖ’дёӢдёҖдёӘпјҲеҪ’зәіеҒҮи®ҫ + жҺЁеҮәдёӢдёҖжӯҘпјү
    • жүҖжңүйӘЁзүҢйғҪдјҡеҖ’дёӢпјҲе‘ҪйўҳеҜ№жүҖжңү $n$ жҲҗз«Ӣпјү

    жӯҘйӘӨ¶

    иҰҒиҜҒжҳҺе‘Ҫйўҳ $P(n)$ еҜ№жүҖжңү $n \geq 1$ жҲҗз«ӢпјҢйҖҡеёёйңҖиҰҒдёӨжӯҘпјҡ

    • з¬¬дёҖжӯҘпјҡйӘҢиҜҒеҹәзЎҖпјҲBase Caseпјү

    йӘҢиҜҒе‘Ҫйўҳ $P(1)$ жҲҗз«ӢгҖӮ

    • з¬¬дәҢжӯҘпјҡеҪ’зәіжӯҘйӘӨпјҲInductive Stepпјү

    еҒҮи®ҫ $P(k)$ жҲҗз«ӢпјҲеҪ’зәіеҒҮи®ҫпјүпјҢиҜҒжҳҺ $P(k+1)$ д№ҹжҲҗз«ӢгҖӮ

    иӢҘдёҠиҝ°дёӨжӯҘйғҪжҲҗз«ӢпјҢеҲҷж №жҚ®ж•°еӯҰеҪ’зәіжі•пјҢ$P(n)$ еҜ№жүҖжңүжӯЈж•ҙж•° $n$ жҲҗз«ӢгҖӮ

    дҫӢйўҳ¶

    зӯүе·®ж•°еҲ—жұӮе’Ңе…¬ејҸ

    з”Ёж•°еӯҰеҪ’зәіжі•иҜҒжҳҺпјҡ

    \[ 1 + 2 + \dots + n = \dfrac{n(n+1)}{2} \]

    жӯҘйӘӨдёҖпјҲеҹәзЎҖжғ…еҶөпјүпјҡеҪ“ $n=1$ ж—¶пјҢ

    \[ 1 = \dfrac{1(1+1)}{2} = 1 \quad \text{вң… жҲҗз«Ӣ} \]

    жӯҘйӘӨдәҢпјҲеҪ’зәіеҒҮи®ҫпјүпјҡ

    еҒҮи®ҫеҜ№жҹҗдёӘ $k$ жҲҗз«Ӣпјҡ

    \[ 1 + 2 + \dots + k = \dfrac{k(k+1)}{2} \]

    иҰҒиҜҒ $k+1$ жҲҗз«ӢпјҢеҚіиҜҒжҳҺ

    \[ 1 + 2 + \dots + k + (k+1) = \dfrac{(k+1)(k+2)}{2} \]

    е·Ұиҫ№ж №жҚ®еҒҮи®ҫеҸҳдёәпјҡ

    \[ \dfrac{k(k+1)}{2} + (k+1) = \dfrac{(k+1)(k + 2)}{2} \]

    еҪ’зәіжҲҗеҠҹпјҒз»“и®әжҲҗз«ӢгҖӮ

    дҫӢйўҳпјҡ$2^n \ge n+1$ дё” $n\geq 1$

    з”Ёж•°еӯҰеҪ’зәіжі•иҜҒжҳҺпјҡ

    \[ 2^n \ge n+1, \quad \text{еҜ№жүҖжңү } n \ge 1 \text{ жҲҗз«Ӣ} \]

    еҹәзЎҖжғ…еҶөпјҡ$n = 1$ ж—¶пјҡ

    \[ 2^1 = 2 \ge 2 = 1 + 1 \quad \text{вң… жҲҗз«Ӣ} \]

    еҪ’зәіеҒҮи®ҫпјҡеҒҮи®ҫ $2^k \ge k+1$

    иҰҒиҜҒжҳҺпјҡ

    \[ 2^{k+1} \ge k + 2 \]

    е·Ұиҫ№пјҡ

    \[ 2^{k+1} = 2 \cdot 2^k \ge 2(k + 1) \]

    еҸӘйңҖиҜҒ $2(k + 1) \ge k + 2$пјҢеҚіпјҡ

    \[ 2k + 2 \ge k + 2 \Rightarrow k \ge 0 \quad \text{вң… жҒ’жҲҗз«Ӣ} \]

    жүҖд»Ҙ $2^{k+1} \ge k + 2$ жҲҗз«ӢгҖӮ

    иҜҒжҳҺе®ҢжҜ•гҖӮ

    жҖ»з»“¶

    • еҝҳи®°йӘҢиҜҒеҹәзЎҖжғ…еҶөпјҡеҝ…йЎ»е…ҲйӘҢиҜҒ $n=1$пјҲжҲ–жҢҮе®ҡиө·зӮ№пјүгҖӮ
    • еҪ’зәіеҒҮи®ҫдҪҝз”Ёй”ҷиҜҜпјҡ еҸӘиғҪеҒҮи®ҫ $P(k)$ жҲҗз«ӢпјҢдёҚеҸҜеҒҮи®ҫ $P(k+1)$гҖӮ
    • жҺЁзҗҶйҖ»иҫ‘дёҚжё…жҷ°пјҡ еҪ’зәіжӯҘйӘӨдёӯйңҖйҖ»иҫ‘дёҘеҜҶең°жҺЁеҮә $P(k+1)$гҖӮ
    • еҜ№йқһж•ҙж•°дҪҝз”ЁеҪ’зәіжі•пјҡ еҪ’зәіжі•еҸӘйҖӮз”ЁдәҺжӯЈж•ҙж•°жҲ–еҸҜжһҡдёҫйӣҶеҗҲгҖӮ

    е°Ҹз»ғд№ ¶

    дҪҝз”Ёж•°еӯҰеҪ’зәіжі•иҜҒжҳҺпјҡиӢҘ $p$ дёәзҙ ж•°пјҢ$\gcd(a, p) = 1$пјҢеҲҷ $a^{p} \equiv a \pmod{p}$гҖӮ

    • йӘҢиҜҒ $a=1$ зҡ„жғ…еҶө

    жҳҫз„¶ $1^p\equiv 1\pmod p$

    еүҚзҪ®иҜҒжҳҺ

    иӢҘ $p$ дёәзҙ ж•°пјҢиҜҒжҳҺеңЁжЁЎ $p$ ж„Ҹд№үдёӢ

    \[ \binom{p}{1}\equiv \binom{p}{2}\equiv \cdots \equiv \binom{p}{p-1}\equiv 0\pmod p \]

    ж №жҚ®з»„еҗҲж•°е…¬ејҸеҸҜеҫ—

    \[ \binom{p}{i}=\dfrac{p!}{i!\times (p-i)!}=\dfrac{(p-i+1)\times (p-i+2)\times \ldots \times p}{1\times 2\times \ldots \times i},\ i\in [1,p-1] \]

    з”ұдәҺ $p$ жҳҜзҙ ж•°пјҢеӣ жӯӨ $1\sim p$ дёӯеҸӘжңү $p\equiv 0\bmod p$гҖӮеӣ жӯӨеҸҜеҫ—

    \[ \binom{p}{i}\equiv 0\bmod p,\ i\in [1,p-1] \]
    • еҪ’зәіеҒҮи®ҫ

    еҒҮи®ҫ $a^p\equiv a\pmod p$ жҲҗз«ӢпјҢйӮЈд№ҲйҖҡиҝҮдәҢйЎ№ејҸе®ҡзҗҶжңү

    \[\begin{aligned} &(a+1)^p\\ =&\sum\limits_{i=0}^p \binom{p}{i} a^{p-i}\cdot 1^i\\ =&a^p+\binom{p}{1}a^{p-1}+\binom{p}{2}a^{p-2}+\cdots +\binom{p}{p-1}a+1 \end{aligned}\]

    з”ұдәҺ $p$ дёәзҙ ж•°пјҢеңЁжЁЎ $p$ ж„Ҹд№үдёӢ

    \[ \binom{p}{1}\equiv \binom{p}{2}\equiv \cdots \equiv \binom{p}{p-1}\equiv 0\pmod p \]

    еӣ жӯӨ $(a+1)^p\equiv a^p+1 \pmod p$гҖӮ

    е°Ҷ $a^p\equiv a\pmod p$ еёҰе…Ҙеҫ— $(a+1)^p\equiv a+1\pmod p$ еҫ—иҜҒгҖӮ

    йҖҶе…ғ¶

    еј•е…Ҙ¶

    • еҗ‘еүҚиө° $3$ жӯҘпјҢеҶҚеҗ‘еҗҺиө° $3$ жӯҘпјҢе°ұеӣһеҲ°дәҶеҺҹзӮ№гҖӮ
    • еҗғдәҶ $5$ йў—зі–пјҢжғі вҖңеӣһеҲ°еҺҹжқҘвҖқпјҢе°ұиҰҒ вҖңдёҚеҗғвҖқ $5$ йў—гҖӮ
    • иҝҷдәӣж“ҚдҪңйғҪжҳҜвҖңдә’зӣёжҠөж¶ҲвҖқзҡ„гҖӮ

    еҠ жі•дёӯзҡ„ йҖҶ

    \[ 5 + (-5) = 0 \]
    • жҲ‘д»¬жҠҠ $-5$ з§°дёә $5$ зҡ„ еҠ жі•йҖҶе…ғгҖӮ
    • еҠ жі•йҖҶе…ғзҡ„дҪңз”ЁжҳҜпјҡи®©з»“жһңеҸҳжҲҗ $0$гҖӮ

    еҰӮжһң $a + b = 0$пјҢжҲ‘д»¬з§° $b$ жҳҜ $a$ зҡ„еҠ жі•йҖҶе…ғгҖӮ

    д№ҳжі•дёӯзҡ„ йҖҶ

    \[ 4 \times \frac{1}{4} = 1 \]
    • $\frac{1}{4}$ жҳҜ $4$ зҡ„д№ҳжі•йҖҶе…ғгҖӮ
    • д№ҳжі•йҖҶе…ғзҡ„дҪңз”ЁжҳҜпјҡи®©з»“жһңеҸҳжҲҗ $1$гҖӮ

    еҰӮжһң $a \cdot b = 1$пјҢжҲ‘д»¬з§° $b$ жҳҜ $a$ зҡ„д№ҳжі•йҖҶе…ғгҖӮ

    е®ҡд№ү¶

    жЁЎж„Ҹд№үдёӢзҡ„йҖҶе…ғ

    иӢҘеӯҳеңЁж•ҙж•° $x$пјҢдҪҝеҫ—пјҡ

    \[ a \cdot x \equiv 1 \pmod{m} \]

    еҲҷз§° $x$ жҳҜ $a$ е…ідәҺжЁЎ $m$ зҡ„д№ҳжі•йҖҶе…ғгҖӮ

    йҖҶе…ғеӯҳеңЁзҡ„жқЎд»¶

    • дёҚжҳҜжҜҸдёӘж•°йғҪжңүйҖҶе…ғпјҒ
    • еҪ“дё”д»…еҪ“ $\gcd(a, m) = 1$пјҢйҖҶе…ғжүҚеӯҳеңЁгҖӮ

    еҚіеҪ“ж»Ўи¶і $\gcd(a, m) = 1$пјҢеҲҷеӯҳеңЁж•ҙж•° $x$пјҢдҪҝеҫ—пјҡ

    \[ a \cdot x \equiv 1 \pmod{m} \]

    йҖҶе…ғзҡ„еӯҳеңЁжҖ§иҜҒжҳҺ¶

    еҸҚиҜҒжі•пјҡ и®ҫ $a$ дёҺ $b$ дёҚдә’зҙ пјҢеҚі $\gcd(a, b) = d \ne 1$пјҢжҲ‘д»¬иҜ•еӣҫиҜҒжҳҺжӯӨж—¶йҖҶе…ғдёҚеӯҳеңЁгҖӮ

    еҒҮи®ҫпјҡ еӯҳеңЁ $x$ дҪҝеҫ—пјҡ

    \[ ax \equiv 1 \pmod{b} \quad \Rightarrow \quad ax = kb + 1 \quad \text{еҜ№дәҺжҹҗдёӘж•ҙж•° } \ k \ \text{жҲҗз«Ӣ} \]

    дёӨиҫ№еҗҢж—¶йҷӨд»Ҙ $d$ еҫ—пјҡ

    \[ \frac{ax}{d} = \frac{kb + 1}{d} \]

    жіЁж„ҸеҲ°пјҡ

    • $d \mid a$ дё” $d \mid b$пјҢжүҖд»Ҙ $\dfrac{ax}{d}$ е’Ң $\dfrac{kb}{d}$ йғҪжҳҜж•ҙж•°пјӣ
    • з„¶иҖҢ $\dfrac{1}{d}$ дёҚжҳҜж•ҙж•°пјҲеӣ дёә $d > 1$пјүгҖӮ

    зҹӣзӣҫпјҡ е·Ұиҫ№дёәж•ҙж•°пјҢеҸіиҫ№еҗ«йқһж•ҙж•°йЎ№ $\frac{1}{d}$пјҒ

    жҖ§иҙЁ¶

    • йҖҶе…ғеңЁжЁЎ $p$ дёӢжҳҜе”ҜдёҖзҡ„гҖӮ

    иҜҒжҳҺжҖқи·Ҝпјҡ иӢҘ $i_1$ е’Ң $i_2$ йғҪжҳҜ $x$ зҡ„йҖҶе…ғпјҢеҲҷпјҡ

    \[ x \cdot i_1 \equiv 1 \pmod{p}, \quad x \cdot i_2 \equiv 1 \pmod{p} \]

    дёӨејҸзӣёзӯүпјҢе·Ұиҫ№зӣёзӯүпјҢеҫ—пјҡ

    \[ x(i_1 - i_2) \equiv 0 \pmod{p} \]

    з”ұдәҺ $\gcd(x, p) = 1$пјҢеҸҜзәҰеҺ» $x$пјҢеҫ—пјҡ

    \[ i_1 \equiv i_2 \pmod{p} \]

    еҚійҖҶе…ғе”ҜдёҖпјҲжЁЎж„Ҹд№үдёӢе”ҜдёҖпјүгҖӮ

    • жЁЎиҙЁж•° $p$ дёӢпјҢ$1 \sim p-1$ дёӯжҜҸдёӘж•°йғҪжңүе”ҜдёҖйҖҶе…ғпјҢдё”дёӨдёӨдёҚеҗҢгҖӮ
      • жЁЎ $p$ дёӢпјҢйҖҶе…ғжһ„жҲҗдёҖдёӘзҪ®жҚўгҖӮ
    • йҖҶе…ғзҡ„йҖҶе…ғжҳҜе…¶жң¬иә«пјҡ
    \[ \text{иӢҘ }\ a^{-1} \equiv x \pmod{p} \text{пјҢеҲҷ }\ x^{-1} \equiv a \pmod{p} \]

    иҜҙжҳҺпјҡ еӣ дёә $a \cdot x \equiv 1$пјҢдёӨиҫ№дә’жҚўеҚіеҫ—з»“и®әгҖӮ

    иғҪе№Ід»Җд№Ҳпјҹ¶

    дҫӢеҰӮпјҡеңЁжЁЎ $7$ ж„Ҹд№үдёӢпјҢжұӮ $3$ зҡ„д№ҳжі•йҖҶе…ғгҖӮ

    • е“ӘдёӘж•°д№ҳд»Ҙ $3$пјҢжЁЎ $7$ еҗҺзӯүдәҺ $1$пјҹ
    • е°қиҜ•пјҡ$3 \times 1 = 3$, $3 \times 2 = 6$, $3 \times 5 = 15 \equiv 1$
    \[ \boxed{3^{-1} \equiv 5 \pmod{7}} \]

    жҚўеҸҘиҜқжқҘиҜҙпјҢеңЁжЁЎ $7$ ж„Ҹд№үдёӢпјҢйҷӨд»Ҙ $3$ е’Ңд№ҳд»Ҙ $5$ зӯүд»·пјҒ

    йҖҶе…ғзҡ„жұӮжі•¶

    еҲҶзұ»¶

    еүҚжҸҗжқЎд»¶пјҡ дёҖдёӘж•° $x$ еңЁжЁЎ $p$ дёӢеӯҳеңЁйҖҶе…ғ $\iff \gcd(x,\ p) = 1$гҖӮ

    и§Јжі•еҲҶзұ»пјҡ

    • й’ҲеҜ№еҚ•дёӘж•°еӯ—зҡ„йҖҶе…ғпјҢеӯҳеңЁдёӨз§Қ $\log$ еӨҚжқӮеәҰеҶ…зҡ„и§Јжі•гҖӮ
      • дёҖз§Қж–№жі•й’ҲеҜ№жЁЎж•°жҳҜеӨ§иҙЁж•°пјҢдё” $x<p$пјҢиҝҷж ·еҸҜд»ҘдҝқиҜҒ $\gcd(x,p)=1$гҖӮ
      • дёҖз§Қж–№жі•еҸӘиҰҒжұӮ $\gcd(x,p)=1$пјҢйңҖиҰҒдҪҝз”Ё exgcd з®—жі•гҖӮ
    • й’ҲеҜ№ $n$ дёӘж•°еӯ—зҡ„йҖҶе…ғпјҢеӯҳеңЁ $O(n)$ зҡ„и§ЈеҶіеҠһжі•гҖӮ

    иҙ№й©¬е°Ҹе®ҡзҗҶ¶

    еҪ“жЁЎж•° $p$ жҳҜиҙЁж•°ж—¶пјҢеҸҜз”Ёиҙ№й©¬е°Ҹе®ҡзҗҶеҝ«йҖҹжұӮйҖҶе…ғгҖӮ

    иҙ№й©¬е°Ҹе®ҡзҗҶпјҡиӢҘ $p$ жҳҜиҙЁж•°пјҢдё” $\gcd(a, p) = 1$пјҢеҲҷпјҡ

    \[ a^{p-1} \equiv 1 \pmod{p} \]

    дёӨиҫ№еҗҢж—¶йҷӨд»Ҙ $a$пјҢеҸҜеҫ—пјҡ

    \[ a^{-1} \equiv a^{p-2} \pmod{p} \]

    з»“и®әпјҡ еҸӘйңҖдҪҝз”Ёеҝ«йҖҹе№ӮжұӮеҮә $a^{p-2} \bmod p$ еҚіеҸҜгҖӮ

    ж—¶й—ҙеӨҚжқӮеәҰпјҡ$O(\log{p})$пјҢйҖӮз”ЁдәҺз»қеӨ§еӨҡж•°жғ…еҶөгҖӮеүҚжҸҗ $p\nmid a$гҖӮ

    long long ksm(long long a, long long b, long long p) 
{
    long long res = 1;
    while (b) 
    {
        if (b & 1) res = res * a % p;
        a = a * a % p;
        b >>= 1;
    }
    return res;
}
long long inv(long long a, long long p) 
{
    return ksm(a, p - 2, p); // иҙ№й©¬е°Ҹе®ҡзҗҶ
}

    еҪ“жЁЎж•°еҫҲеӨ§ж—¶пјҢжҷ®йҖҡд№ҳжі•еҸҜиғҪеҜјиҮҙ $\texttt{long long}$ жәўеҮәгҖӮ

    еҸҜдҪҝз”Ё $\texttt{__int128}$ зұ»еһӢеҒҡдёӯй—ҙиҪ¬жҚўпјҢйҒҝе…ҚжәўеҮәгҖӮ

    жү©еұ•ж¬§еҮ йҮҢеҫ—¶

    иҰҒеңЁжЁЎ $p$ ж„Ҹд№үдёӢжұӮ $x^{-1}$пјҢеҚіжүҫеҲ°дёҖдёӘж•ҙж•° $x^{-1}$ ж»Ўи¶іпјҡ

    \[ x \cdot x^{-1} \equiv 1 \pmod{p} \]

    е°ҶеҗҢдҪҷејҸиҪ¬еҢ–дёәзӯүејҸпјҢжңүпјҡ

    \[ x \cdot x^{-1} + p \cdot k = 1 \quad (k \in \mathbb{Z}) \]

    иҝҷжҳҜдёҖдёӘеҪўеҰӮпјҡ

    \[ ax + by = \gcd(a, b) \]

    зҡ„ зәҝжҖ§дёҚе®ҡж–№зЁӢпјҢеҸҜйҖҡиҝҮ жү©еұ•ж¬§еҮ йҮҢеҫ—з®—жі• и§Јеҫ—дёҖз»„ж•ҙж•°и§Ј $(x^{-1}, k)$гҖӮ

    жіЁж„Ҹпјҡ иӢҘ $\gcd(x, p) \ne 1$пјҢж–№зЁӢж— ж•ҙж•°и§ЈпјҢйҖҶе…ғд№ҹдёҚеӯҳеңЁгҖӮ

    ж—¶й—ҙеӨҚжқӮеәҰпјҡ$O(\log{p})$

    long long exgcd(long long a, long long b, long long &x, long long &y) 
{
    if (b == 0) 
    {
        x = 1, y = 0;
        return a;
    }
    long long d = exgcd(b, a % b, x, y);
    long long t = x; 
    x = y; 
    y = t - a / b * y;
    return d;
}
long long inv(long long a, long long b) 
{
    long long x, y;
    exgcd(a, b, x, y); 
    x = (x % b + b) % b; 
    return x;
}

    зәҝжҖ§йў„еӨ„зҗҶйҖҶе…ғ¶

    й—®йўҳиғҢжҷҜпјҡ иӢҘжЁЎж•° $p$ жҳҜиҙЁж•°пјҢдё”жҲ‘д»¬йңҖиҰҒйў‘з№ҒжұӮ $1\sim n$пјҲ$n< p$пјү дёӯжҜҸдёӘж•°зҡ„йҖҶе…ғпјҢдҪҝз”Ёеҝ«йҖҹе№Ӯдјҡжңү $O(n\log p)$ зҡ„еӨҚжқӮеәҰгҖӮ

    зӣ®ж Үпјҡ дҪҝз”ЁзәҝжҖ§йҖ’жҺЁпјҢеңЁ $O(n)$ ж—¶й—ҙеҶ…йў„еӨ„зҗҶжүҖжңүйҖҶе…ғгҖӮ

    йҖ’жҺЁе…¬ејҸпјҡ

    \[ \boxed{ \texttt{inv[1]} = 1,\quad \texttt{inv[i]} = (p - \lfloor \frac{p}{i} \rfloor) \cdot \texttt{inv}[p \bmod i] \bmod p } \]

    жҲ–иҖ…еҶҷдҪңпјҡ

    \[ \texttt{inv[i]} \equiv - \left\lfloor \frac{p}{i} \right\rfloor \cdot \texttt{inv}[p \bmod i] \pmod{p} \]

    ж—¶й—ҙеӨҚжқӮеәҰпјҡ $O(n)$

    иҜҒжҳҺ

    е·ІзҹҘпјҡ$p = ki + r$пјҢе…¶дёӯ $i\in [1, n]$ еҚі

    \[ ki + r \equiv 0 \pmod{p} \]

    е°ҶзӯүејҸдёӨиҫ№еҗҢж—¶д№ҳд»Ҙ $i^{-1} \cdot r^{-1}$пјҢеҫ—еҲ°пјҡ

    \[ k \cdot r^{-1} + i^{-1} \equiv 0 \pmod{p} \Rightarrow i^{-1} \equiv -k \cdot r^{-1} \pmod{p} \]

    з”ұжӯӨеҫ—еҮәпјҡ

    \[ \boxed{ \texttt{inv[i]} \equiv - \left\lfloor \frac{p}{i} \right\rfloor \cdot \texttt{inv}[p \bmod i] \pmod{p} } \]

    зӣҙи§ӮзҗҶи§Јпјҡ еҲ©з”Ёиҫғе°Ҹж•°зҡ„йҖҶе…ғйҖ’жҺЁеҮәиҫғеӨ§зҡ„йҖҶе…ғгҖӮ

    constexpr int N = 3e6 + 5;
int inv[N], p;
void init(int n) 
{
    inv[1] = 1;
    for (int i = 2; i <= n; ++i) 
    {
        inv[i] = 1ll * (p - p / i) * inv[p % i] % p;
    }
}

    иҜҘж–№жі•иҰҒжұӮ $p$ жҳҜиҙЁж•°пјҢдҝқиҜҒ $n<p$ йғҪжңүйҖҶе…ғпјӣ

    й—®йўҳиғҢжҷҜпјҡ з»ҷе®ҡ $n$ дёӘж•ҙж•° $a_1,a_2,\ldots,a_n$пјҢжұӮе®ғд»¬еңЁжЁЎ $p$ дёӢзҡ„йҖҶе…ғгҖӮдҝқиҜҒжүҖжңү $a_i < p$гҖӮдёәйҒҝе…Қиҫ“еҮәиҝҮеӨҡпјҢиҜ·жұӮеҮә $\sum\limits_{i=1}^n \dfrac{k^i}{a_i}$ зҡ„еҖјгҖӮе…¶дёӯ $n\leq 5\times 10^6$гҖӮ

    ж–№жі•пјҡ

    • йў„еӨ„зҗҶ $k[i]$ иЎЁзӨә $k^i$пјҢж №жҚ® $k^i=k^{i-1}\times k$ йҖ’жҺЁжұӮеҫ—еҚіеҸҜгҖӮж—¶й—ҙеӨҚжқӮеәҰпјҡ$O(n)$гҖӮ

    еҲҷй—®йўҳзҡ„йҡҫзӮ№еңЁдәҺеҰӮдҪ•зәҝжҖ§жұӮеҮә $a_i$ зҡ„йҖҶе…ғгҖӮ

    еҲ©з”ЁеүҚзјҖз§ҜйҖҶе…ғзҡ„йҖ’жҺЁе…ізі»жұӮйҖҶе…ғгҖӮ

    и®ҫ $s_i = a_1\times a_2\times \ldots \times a_i \pmod p$пјҢе…¶дёӯ $s_0=1$гҖӮ

    еҲҷжңүйҖҶе…ғе…ізі»пјҡ

    \[ s_i^{-1} = s_{i+1}^{-1} \times a_{i+1} \pmod p \]

    иҝҷжқҘжәҗдәҺеҲҶж•°зәҰеҲҶпјҡеҲҶеӯҗеҲҶжҜҚеҗҢд№ҳйҷӨдёҖдёӘ $a_{i+1}$гҖӮ

    $\dfrac{1}{s_i}=\dfrac{1}{s_{i+1}}\times a_{i+1}$гҖӮ

    $s_i=a_1\times a_2\times \ldots \times a_i \pmod p$гҖӮ

    $s_{i+1}=a_1\times a_2\times \ldots \times a_i\times a_{i+1} \pmod p$гҖӮ

    жңүдәҶ $s_i$ д»ҘеҸҠ $s_{i}^{-1}$ д»ҘеҗҺпјҢе°ұеҸҜд»ҘжұӮи§Ј $a_i^{-1}$ дәҶгҖӮ

    ж №жҚ® $a_i=\dfrac{s_{i-1}}{s_i}=s^{i-1}\times s_i^{-1}$ еҸҜд»ҘжӯЈеёёжұӮеҮәжҜҸдёҖйЎ№гҖӮ

    е…·дҪ“жөҒзЁӢ

    • и®Ўз®— $k^i$ дҝқеӯҳеҲ°ж•°з»„ $k[i]$ дёӯгҖӮ
    • и®Ўз®—еүҚзјҖз§Ҝ $s_i = s_{i-1} \times a_i \pmod p$ дҝқеӯҳеҲ°ж•°з»„ $s[i]$ дёӯгҖӮ
    • жұӮеҮәжҖ»з§ҜйҖҶе…ғ $s_n^{-1}$ дҝқеӯҳеҲ°ж•°з»„ $invs[n]$ дёӯпјҢжң¬йўҳеҸҜд»Ҙз”Ёиҙ№й©¬е°Ҹе®ҡзҗҶжұӮеҮәгҖӮ
    • йҖҶеәҸйҖ’жҺЁи®Ўз®— $i\in [n-1,0]$ ж—¶е…¶дҪҷ $s_i^{-1}$ дҝқеӯҳеҲ°ж•°з»„ $invs[n-1]\sim invs[0]$ дёӯгҖӮ

    #include <bits/stdc++.h>
using namespace std;
constexpr int N = 5e6 + 5; 
int n, p, K, a[N], k[N], s[N], invs[N];
int read()
{
    int s = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9')
    {
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9')
    {
        s = s * 10 + ch - '0';
        ch = getchar();
    }
    return s * f;
}
long long ksm(long long a, long long b)
{
    long long res = 1;
    while (b)
    {
        if (b & 1) res = res * a % p;
        a = a * a % p;
        b >>= 1;
    }
    return res; 
}
int main()
{
    k[0] = 1;
    n = read(), p = read(), K = read();
    for (int i = 1; i <= n; i++) k[i] = 1ll * k[i - 1] * K % p;
    s[0] = 1; 
    for (int i = 1; i <= n; i++)
    {
        a[i] = read();
        s[i] = 1ll * s[i - 1] * a[i] % p;
    }
    invs[n] = ksm(s[n], p - 2);// еҲқе§ӢеҢ– s[n] зҡ„йҖҶе…ғ 
    for (int i = n - 1; i >= 0; i--)
        invs[i] = 1ll * invs[i + 1] * a[i + 1] % p; 
    long long ans = 0;
    for (int i = 1; i <= n; i++)
    {
        long long inva = 1ll * s[i - 1] * invs[i] % p; 
        ans = (ans + 1ll * k[i] * inva % p) % p;
    }
    cout << ans;
    return 0;
}
