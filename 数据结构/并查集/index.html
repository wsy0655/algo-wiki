    #include <bits/stdc++.h>
using namespace std;
const int N = 1.5e5 + 5;
int n, m, fa[N], ans;
int find(int x)
{
    return x == fa[x] ? fa[x] : fa[x] = find(fa[x]);
}
void merge(int x, int y)
{
    x = find(x), y = find(y);
    if (x == y) return ;
    fa[x] = y;
}
int main()
{
    ios::sync_with_stdio(false), cin.tie(0);
    cin >> n >> m;
    for (int i = 1; i <= n + 1; i++)
        fa[i] = i;
    int ans = n;
    while (m--)
    {
        int l, r;
        cin >> l >> r;
        for (int j = find(l); j <= r; j = find(j + 1))
        {
            fa[j] = j + 1;
            ans--;
        }
        cout << ans << '\n';
    }
    return 0;
}
