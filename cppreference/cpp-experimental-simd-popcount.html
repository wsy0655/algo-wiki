std::experimental::popcount

< cpp‎ | experimental‎ | simd
在标头 <experimental/simd> 定义
template< class T, class Abi >
int popcount( const simd_mask<T, Abi>& k ) noexcept;
 (并行 TS v2)

返回掩码 k 中的 true 值数量。

参数

k - 要运用归约的 simd_mask

返回值

范围 [​0​, simd_size_v<T, Abi>) 中的一个 int 值。

示例

#include <cstddef>
#include <experimental/simd>
#include <iostream>
namespace stdx = std::experimental;
 
template<typename Abi>
int count_zeros(stdx::simd<int, Abi> v)
{
    return stdx::popcount(v == 0); // v == 0 returns a simd_mask
}
 
void println(auto rem, auto const v)
{
    std::cout << rem << ": ";
    for (std::size_t i = 0; i != v.size(); ++i)
        std::cout << v[i] << ' ';
}
 
int main()
{
    stdx::simd<int> x{8};
    println("x", x);
    std::cout << "  zeros: " << count_zeros(x) << '\n';
 
    x[3] = x[1] = false;
    println("x", x);
    std::cout << "  zeros: " << count_zeros(x) << '\n';
}

可能的输出：

x: 8 8 8 8   zeros: 0
x: 8 0 8 0   zeros: 2
