std::unique_ptr<T,Deleter>::get

< cpp‎ | memory‎ | unique ptr
pointer get() const noexcept;
 (C++11 起)
(constexpr since C++23)

返回指向被管理对象的指针，如果无被管理对象，则为 nullptr。

参数

（无）

返回值

指向被管理对象的指针，无被管理对象，则为 nullptr。

示例

#include <iomanip>
#include <iostream>
#include <memory>
#include <string>
#include <utility>
 
class Res
{
    std::string s;
 
public:
    Res(std::string arg) : s{std::move(arg)}
    {
        std::cout << "Res::Res(" << std::quoted(s) << ");\n";
    }
 
    ~Res()
    {
        std::cout << "Res::~Res();\n";
    }
 
private:
    friend std::ostream& operator<<(std::ostream& os, Res const& r)
    {
        return os << "Res { s = " << std::quoted(r.s) << "; }";
    }
};
 
int main()
{
    std::unique_ptr<Res> up(new Res{"Hello, world!"});
    Res* res = up.get();
    std::cout << *res << '\n';
}

输出：

Res::Res("Hello, world!");
Res { s = "Hello, world!"; }
Res::~Res();
