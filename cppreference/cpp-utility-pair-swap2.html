std::swap(std::pair)

< cpp‎ | utility‎ | pair
在标头 <utility> 定义
(1)
template< class T1, class T2 >

void swap( std::pair<T1,T2>& x, std::pair<T1,T2>& y )

    noexcept(/* 见下文 */);
 (C++11 起)
(C++20 前)
template< class T1, class T2 >

constexpr void swap( std::pair<T1,T2>& x, std::pair<T1,T2>& y )

    noexcept(/* 见下文 */);
 (C++20 起)
template< class T1, class T2 >

constexpr void swap( const std::pair<T1,T2>& x, const std::pair<T1,T2>& y )

    noexcept(/* 见下文 */);
 (2) (C++23 起)

交换 x 与 y 的内容。等价于 x.swap(y)。

1) 此重载只有在 std::is_swappable_v<T1> && std::is_swappable_v<T2> 为 true 时才会参与重载决议。
2) 此重载只有在 std::is_swappable_v<const T1> && std::is_swappable_v<const T2> 为 true 时才会参与重载决议。
 (C++17 起)

参数

x, y - 要交换内容的对偶

返回值

（无）

异常

noexcept 说明：  
noexcept(noexcept(x.swap(y)))

示例

#include <iostream>
#include <utility>
 
int main()
{
    auto p1 = std::make_pair(10, 3.14);
    auto p2 = std::pair(12, 1.23); // CTAD, C++17 起
 
    auto print_p1_p2 = [&](auto msg) {
        std::cout << msg
                  << "p1 = {" << std::get<0>(p1)
                  << ", "     << std::get<1>(p1) << "}, "
                  << "p2 = {" << std::get<0>(p2)
                  << ", "     << std::get<1>(p2) << "}\n";
    };
 
    print_p1_p2("p1.swap(p2)  前: ");
    p1.swap(p2);
    print_p1_p2("p1.swap(p2)  后: ");
    std::swap(p1, p2);
    print_p1_p2("swap(p1, p2) 后: ");
}

输出：

p1.swap(p2)  前: p1 = {10, 3.14}, p2 = {12, 1.23}
p1.swap(p2)  后: p1 = {12, 1.23}, p2 = {10, 3.14}
swap(p1, p2) 后: p1 = {10, 3.14}, p2 = {12, 1.23}
