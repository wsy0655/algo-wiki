std::experimental::simd<T,Abi>::operator++, std::experimental::simd<T,Abi>::operator--

< cpp‎ | experimental‎ | simd‎ | simd
simd& operator++() noexcept;
 (1) (并行 TS v2)
simd operator++( int ) noexcept;
 (2) (并行 TS v2)
simd& operator--() noexcept;
 (3) (并行 TS v2)
simd operator--( int ) noexcept;
 (4) (并行 TS v2)

对 simd 的每个元素运用自增或自减运算符。

1) 对 simd 中的所有值增加 1 并返回其自身的引用。
2) 对 simd 中的所有值增加 1 并返回其自身操作前的副本。
3) 对 simd 中的所有值减少 1 并返回其自身的引用。
4) 对 simd 中的所有值减少 1 并返回其自身操作前的副本。

示例

#include <cstddef>
#include <experimental/simd>
#include <iostream>
namespace stdx = std::experimental;
 
void print(auto rem, auto const& a)
{
    std::cout << rem << ": ";
    for (std::size_t i{}; i != std::size(a); ++i)
        std::cout << a[i] << ' ';
    std::cout << '\n';
}
 
int main()
{
    stdx::native_simd<int> p = -2;
    print('p', p);
 
    ++p;
    print('p', p);
 
    auto q = p--;
    print('p', p);
    print('q', q);
}

可能的输出：

p: -2 -2 -2 -2
p: -1 -1 -1 -1 
p: -2 -2 -2 -2
q: -1 -1 -1 -1
