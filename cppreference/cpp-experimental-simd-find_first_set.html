std::experimental::find_first_set, std::experimental::find_last_set

< cpp‎ | experimental‎ | simd
在标头 <experimental/simd> 定义
template< class T, class Abi >
int find_first_set( const simd_mask<T, Abi>& k );
 (1) (并行 TS v2)
template< class T, class Abi >
int find_last_set( const simd_mask<T, Abi>& k );
 (2) (并行 TS v2)
1) 返回使得 k[i] 为 true 的最小索引 i。
2) 返回使得 k[i] 为 true 的最大索引 i。

如果 any_of(k) 为 false 则其行为未定义。

参数

k - 要对其运用归约的 simd_mask

返回值

范围 [​0​, simd_size_v<T, Abi>) 中的一个 int 值。

示例

#include <cstddef>
#include <experimental/simd>
#include <iostream>
 
namespace stdx = std::experimental;
 
template<typename Abi>
int find(stdx::simd_mask<Abi> const& v)
{
    if (stdx::any_of(v))
        return find_first_set(v);
    return -1;
}
 
int main()
{
    stdx::simd_mask<short> a{0};
    a[2] = a[a.size() - 2] = 1;
 
    for (std::size_t i = 0; i < a.size(); ++i)
        std::cout << a[i] << ' ';
    std::cout << '\n';
 
    std::cout << "find_first_set: " << stdx::find_first_set(a) << '\n';
    std::cout << "find_last_set: " << stdx::find_last_set(a) << '\n';
    std::cout << "find: " << find(a) << '\n';
    a[2] = 0;
    std::cout << "find: " << find(a) << '\n';
    a[a.size() - 2] = 0;
    std::cout << "find: " << find(a) << '\n';
}

可能的输出：

find_first_set: 2
find_last_set: 6
find: 2
find: 6
find: -1
