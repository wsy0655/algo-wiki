std::chrono::file_clock::now

< cpp‎ | chrono‎ | file clock

static std::chrono::time_point<std::chrono::file_clock> now() noexcept;
 (C++20 起)


返回表示当前时间的时间点。

返回值

表示当前时间的时间点。

示例

#include <chrono>
#include <cstddef>
#include <iomanip>
#include <iostream>
#include <numeric>
#include <vector>
 
volatile int sink; // 避免优化
 
void do_some_work(std::size_t size)
{
    std::vector<int> v(size, 42);
    sink = std::accumulate(v.begin(), v.end(), 0); // 确保副作用
}
 
int main()
{
    std::cout << std::fixed << std::setprecision(9) << std::left;
    for (auto size{1ull}; size < 1000'000'000ull; size *= 100)
    {
        const auto start = std::chrono::file_clock::now();
        do_some_work(size);
        const auto end = std::chrono::file_clock::now();
        const std::chrono::duration<double> diff = end - start;
        std::cout << "start = " << start << "; end = " << end << ";\n";
        std::cout << "diff = " << diff << "; size = " << size << '\n';
    }
}
