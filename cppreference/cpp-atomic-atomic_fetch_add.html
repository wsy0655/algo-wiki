std::atomic_fetch_add, std::atomic_fetch_add_explicit

< cpp‎ | atomic
在标头 <atomic> 定义
template< class T >

T atomic_fetch_add( std::atomic<T>* obj,

                    typename std::atomic<T>::difference_type arg ) noexcept;
 (1) (C++11 起)
template< class T >

T atomic_fetch_add( volatile std::atomic<T>* obj,

                    typename std::atomic<T>::difference_type arg ) noexcept;
 (2) (C++11 起)
template< class T >

T atomic_fetch_add_explicit( std::atomic<T>* obj,
                             typename std::atomic<T>::difference_type arg,

                             std::memory_order order ) noexcept;
 (3) (C++11 起)
template< class T >

T atomic_fetch_add_explicit( volatile std::atomic<T>* obj,
                             typename std::atomic<T>::difference_type arg,

                             std::memory_order order ) noexcept;
 (4) (C++11 起)

进行原子加法。原子地将 arg 加到 obj 指向的值，并返回 obj 先前保有的值。如同执行下列内容一般进行运算：

1,2) obj->fetch_add(arg)
3,4) obj->fetch_add(arg, order)

如果 std::atomic<T> 没有 fetch_add 成员（此成员仅对 bool 以外的整数、浮点(C++20 起)和指针类型提供），那么程序非良构。

参数

obj - 指向要修改的原子对象的指针
arg - 要加到存储于原子对象的值的值
order - 内存同步定序

返回值

*obj 的修改顺序中，紧接此函数生效之前的值。

示例

单写者/多读者锁能用 std::atomic_fetch_add 制作。注意此过分简化的实现不是免锁的。

#include <atomic>
#include <chrono>
#include <iostream>
#include <string>
#include <thread>
#include <vector>
 
using namespace std::chrono_literals;
 
// cnt 的含义：
//  5：读者和写者在竞争中。无活跃读者或写者。
//  4...0：有 1...5 个读者活跃。写者被阻塞。
// -1：写者赢得竞争。读者被阻塞。
 
const int N = 5; // 允许四个共时读者
std::atomic<int> cnt(N);
 
std::vector<int> data;
 
void reader(int id)
{
    for(;;)
    {
        // 锁定
        while (std::atomic_fetch_sub(&cnt, 1) <= 0)
            std::atomic_fetch_add(&cnt, 1);
 
        // 读取
        if (!data.empty())
            std::cout << ("读者 " + std::to_string(id) +
                          " 看见 " + std::to_string(*data.rbegin()) + '\n');
        if (data.size() == 25)
            break;
 
        // 解锁
        std::atomic_fetch_add(&cnt, 1);
 
        // 暂停
        std::this_thread::sleep_for(1ms);
    }
}
 
void writer()
{
    for(int n = 0; n < 25; ++n)
    {
        // 锁定
        while (std::atomic_fetch_sub(&cnt, N + 1) != N)
            std::atomic_fetch_add(&cnt, N + 1);
 
        // 写入
        data.push_back(n);
        std::cout << "写者写入 " << n << '\n';
 
        // 解锁
        std::atomic_fetch_add(&cnt, N + 1);
 
        // 暂停
        std::this_thread::sleep_for(1ms);
    }
}
 
int main()
{
    std::vector<std::thread> v;
    for (int n = 0; n < N; ++n)
        v.emplace_back(reader, n);
    v.emplace_back(writer);
 
    for (auto& t : v)
        t.join();
}

输出：

写者写入 0
读者 2 看见 0
读者 3 看见 0
读者 1 看见 0
<...>
读者 2 看见 24
读者 4 看见 24
读者 1 看见 24
