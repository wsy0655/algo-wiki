    иӣҮеҪўж–№йҳө

    йўҳзӣ®з®Җиҝ°¶

    • з»ҷе®ҡжӯЈж•ҙж•° $n$пјҢжһ„йҖ дёҖдёӘ $n \times n$ зҡ„ж–№йҳөгҖӮ
    • д»Һе·ҰдёҠи§’ејҖе§ӢпјҢжҢүйЎәж—¶й’Ҳж–№еҗ‘еЎ«е…Ҙж•°еӯ— $1$ еҲ° $n^2$гҖӮ
    • иҫ“еҮәж јејҸиҰҒжұӮпјҡжҜҸдёӘж•°еӯ—еҚ  $3$ дёӘеӯ—з¬Ұе®ҪпјҢе·Ұдҫ§з©әж јиЎҘйҪҗгҖӮ

    е…ій”®зӮ№еҲҶжһҗ¶

    1. еЎ«е……йЎәеәҸпјҡ
      жҢүйЎәж—¶й’Ҳж–№еҗ‘еЎ«е……пјҢеҚідҫқж¬ЎдёәеҸігҖҒдёӢгҖҒе·ҰгҖҒдёҠеӣӣдёӘж–№еҗ‘еҫӘзҺҜгҖӮ

    2. иҫ№з•ҢеҲӨж–ӯпјҡ
      еҪ“дёӢдёҖжӯҘдҪҚзҪ®и¶Ҡз•ҢжҲ–иҖ…иҜҘдҪҚзҪ®е·Іиў«еЎ«е……ж—¶пјҢйңҖиҰҒж”№еҸҳж–№еҗ‘гҖӮ

    3. зҹ©йҳөзҠ¶жҖҒз»ҙжҠӨпјҡ
      з»ҙжҠӨдёҖдёӘдәҢз»ҙж•°з»„ a з”ЁжқҘеӯҳж”ҫеЎ«е…Ҙзҡ„ж•°еӯ—пјҢеҲқе§ӢеҢ–дёә $0$ иЎЁзӨәжңӘеЎ«гҖӮ

    4. еҲқе§ӢжқЎд»¶пјҡ
      иө·зӮ№дёә (1, 1)пјҢеҲқе§Ӣж–№еҗ‘еҗ‘еҸігҖӮ

    е®һзҺ°жӯҘйӘӨ¶

    • еҲқе§ӢеҢ–зҹ©йҳөеӨ§е°Ҹдёә $n \times n$пјҢжүҖжңүе…ғзҙ дёә $0$гҖӮ
    • и®ҫзҪ®ж–№еҗ‘ж•°з»„ dx[4] = {0, 1, 0, -1}, dy[4] = {1, 0, -1, 0}пјҢеҲҶеҲ«д»ЈиЎЁеҸігҖҒдёӢгҖҒе·ҰгҖҒдёҠзҡ„еҸҳеҢ–йҮҸгҖӮ
    • и®ҫеҪ“еүҚеқҗж Ү (x, y) = (1, 1)пјҢеҪ“еүҚж–№еҗ‘ dir = 0гҖӮ
    • дҫқж¬ЎеЎ«е…Ҙж•°еӯ— num д»Һ $1$ еҲ° $n^2$гҖӮ
      • еЎ«е…Ҙ a[x][y] = num
      • и®Ўз®—дёӢдёҖжӯҘеқҗж Ү (nx, ny) = (x + dirs[d][0], y + dirs[d][1])
      • еҰӮжһң (nx, ny) и¶Ҡз•ҢпјҲnx < 1 || nx > n || ny < 1 || ny > nпјүжҲ– a[nx][ny] != 0пјҢиҜҙжҳҺйңҖжҚўж–№еҗ‘пјҡ
      • d = (d + 1) % 4
      • йҮҚж–°и®Ўз®— (nx, ny)гҖӮ
    • жӣҙж–°еҪ“еүҚеқҗж Үдёә (nx, ny)гҖӮ
    • еЎ«е……е®ҢжҲҗеҗҺпјҢж јејҸеҢ–иҫ“еҮәзҹ©йҳөпјҡ
      • жҜҸдёӘж•°еӯ—еҸіеҜ№йҪҗпјҢеҚ  $3$ дёӘеӯ—з¬Ұе®ҪпјҢз”Ёз©әж јиЎҘйҪҗгҖӮ
      • ж•°еӯ—д№Ӣй—ҙд»Ҙз©әж јеҲҶйҡ”пјҢжҜҸиЎҢжҚўиЎҢгҖӮ

    зӨәдҫӢд»Јз Ғ¶

    #include <bits/stdc++.h>
#define int long long
using namespace std;
int n, a[15][15];
int dx[4] = {0, 1, 0, -1};
int dy[4] = {1, 0, -1, 0};

signed main()
{
    cin >> n;
    int x = 1, y = 1, dir = 0;
    for (int num = 1; num <= n * n; num++)
    {
        a[x][y] = num;
        if (num == n * n) // еЎ«дәҶжңҖеҗҺдёҖдёӘе°ұз»“жқҹйҒҝе…Қз»§з»ӯж— йҷҗеҫӘзҺҜдёӢеҺ»
        {
            break;
        }
        int nx = x + dx[dir], ny = y + dy[dir];
        // ж— жі•еЎ«е……
        while (nx < 1 || nx > n || ny < 1 || ny > n || a[nx][ny] != 0)
        {
            dir = (dir + 1) % 4;
            nx = x + dx[dir];
            ny = y + dy[dir];
        }
        x = nx;
        y = ny;
    }
    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= n; j++)
        {
            cout << setw(3) << a[i][j];
        }
        cout << "\n";
    }
    return 0;
}
