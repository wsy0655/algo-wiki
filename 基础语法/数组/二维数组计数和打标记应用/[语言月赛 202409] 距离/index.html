    жҖқи·ҜжҸҸиҝ°¶

    • з»ҙжҠӨдёҖдёӘдәҢз»ҙж•°з»„ d[105][105]пјҢз”ЁжқҘеӯҳеӮЁжҜҸдёӘеҗҢеӯҰ $p$ еҜ№еҗҢеӯҰ $q$ зҡ„еҪ“еүҚеҘҪж„ҹеәҰпјҢеҲқе§Ӣе…ЁйғЁдёә $0$гҖӮ
    • з»ҙжҠӨдёҖдёӘеҸҳйҮҸ mxпјҢи®°еҪ•еҪ“еүҚжүҖжңүеҘҪж„ҹеәҰдёӯзҡ„жңҖеӨ§еҖјпјҢеҲқе§Ӣдёә $0$гҖӮ
    • еҜ№дәҺжҜҸдёҖд»¶дәӢжғ…пјҡ
    • ж №жҚ®ж“ҚдҪңзұ»еһӢ opпјҡ
      • иӢҘ op == 1пјҢжү§иЎҢеҠ жі•пјҡd[a][b] += cгҖӮ
      • иӢҘ op == 2пјҢжү§иЎҢеҮҸжі•пјҡd[a][b] -= cгҖӮ
    • жӣҙж–°еҗҺпјҢжЈҖжҹҘ d[a][b] жҳҜеҗҰеӨ§дәҺ mxпјҢиӢҘжҳҜеҲҷжӣҙж–° mxгҖӮ
    • жіЁж„ҸпјҢеҮҸжі•еҸҜиғҪдјҡеҜјиҮҙд№ӢеүҚзҡ„жңҖеӨ§еҖјиў«еҮҸе°ҸпјҢеӣ жӯӨйңҖиҰҒйҮҚж–°и®Ўз®—жңҖеӨ§еҖјпјҡ
      • дёҖз§Қз®ҖеҚ•дҪҶж•ҲзҺҮиҫғдҪҺзҡ„ж–№ејҸжҳҜжҜҸж¬Ўж“ҚдҪңеҗҺйҒҚеҺҶж•ҙдёӘ d ж•°з»„пјҢжүҫеҮәеҪ“еүҚжңҖеӨ§еҖјгҖӮ
      • з”ұдәҺ $n \leq 100$пјҢжңҖеӨҡ $100 \times 100 = 10,000$ дёӘе…ғзҙ пјҢдё” $m \leq 100$пјҢиҝҷз§ҚеҒҡжі•жҳҜеҸҜиЎҢзҡ„гҖӮ
    • иҫ“еҮәжҜҸж¬Ўж“ҚдҪңеҗҺзҡ„ mxгҖӮ

    е®һзҺ°зӨәдҫӢ¶

    #include <bits/stdc++.h>
using namespace std;
int d[105][105];
int main() 
{
    int n, m;
    cin >> n >> m;
    while (m--)
    {
        int op, a, b, c;
        cin >> op >> a >> b >> c;
        if (op == 1) d[a][b] += c;
        else d[a][b] -= c;
        int mx = 0;
        // йҮҚж–°йҒҚеҺҶжүҖжңүдҪҚзҪ®жұӮжңҖеӨ§еҖј
        for (int i = 1; i <= n; i++)
            for (int j = 1; j <= n; j++)
                mx = max(mx, d[i][j]);
        cout << mx << "\n";
    }
    return 0;
}
