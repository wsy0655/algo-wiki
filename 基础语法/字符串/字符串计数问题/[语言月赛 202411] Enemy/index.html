using namespace std;

// е®ҡд№үжңҖеӨ§зҹ©йҳөз»ҙеәҰеёёйҮҸ
constexpr int N = 1e3 + 5;

int n, m;         // n иЎҢпјҢm еҲ—
char a[N][N];     // жҲҳеңәең°еӣҫзҹ©йҳө

int main()
{
    // иҫ“е…Ҙзҹ©йҳөеӨ§е°Ҹ
    cin >> n >> m;

    // иҫ“е…ҘжҲҳеңәеұҖеҠҝпјҢжіЁж„Ҹд»Һ 1 ејҖе§Ӣзј–еҸ·пјҢдҫҝдәҺеӨ„зҗҶиҫ№з•Ң

        if (y) // жңүеҹҺеўҷпјҢеҸҢж–№ж’ӨйҖҖ

                a[i][j] = '#';
            // еҸідҫ§ numb дёӘ B еЈ«е…өпјҢд»ҺеҸіиҫ№еҗ‘е·ҰжҺ’еҲ—
            for (int j = m; j >= m - numb + 1; j--)
                a[i][j] = 'B';
            // еҹҺеўҷеҸіиҫ№еӨҡдҪҷзҡ„дҪҚзҪ®зҪ®дёә #
            for (int j = y + 1; j <= m - numb; j++)

        {

            }

    for (int i = 1; i <= n; i++)

        }

    }
