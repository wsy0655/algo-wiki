    • ж—¶й—ҙеӨҚжқӮеәҰпјҡ$O(n!\times n)$

    дҫӢйўҳдәҢпјҡзҒ«жҳҹдәә¶

    йўҳж„Ҹпјҡз»ҷе®ҡ $N$ е’Ң $M$пјҢд»ҘеҸҠдёҖдёӘй•ҝеәҰдёә $N$ зҡ„еҲқе§ӢжҺ’еҲ—пјҢд»ЈиЎЁзҒ«жҳҹдәәеҪ“еүҚзҡ„ ж•°гҖӮжҢүеӯ—е…ёеәҸеҗ‘еҗҺз§»еҠЁ $M$ жӯҘ(и°ғз”Ё next_permutation $M$ ж¬Ў)пјҢиҫ“еҮәжңҖз»ҲжҺ’еҲ—гҖӮ

    жҖқи·ҜпјҡзӣҙжҺҘжү§иЎҢ M ж¬Ў next_permutation(a + 1, a + N + 1)гҖӮ

    дҫӢйўҳдёүпјҡзҺҜжёёеҹҺеёӮ¶

    йўҳж„Ҹпјҡ$N$ еә§еҹҺеёӮпјҢж—…иЎҢж—¶й—ҙзҹ©йҳө T[N][N]пјҢд»ҺеҹҺеёӮ $1$ еҮәеҸ‘жҒ°еҘҪи®ҝй—®жүҖжңүе…¶д»–еҹҺеёӮдёҖж¬ЎеҶҚиҝ”еӣһ $1$пјҢй—®жңүеӨҡе°‘жқЎи·Ҝеҫ„жҖ»ж—¶й—ҙжҒ°еҘҪжҳҜ $K$гҖӮ

    жҖқи·Ҝпјҡ

    е®ҡд№ү $p[N]$ з»ҙжҠӨеҹҺеёӮзј–еҸ·пјҢеҲқе§ӢеҢ– $p[i]=i$гҖӮз”ұдәҺжңҖеҗҺиҝҳиҰҒеӣһеҲ°еҹҺеёӮ $1$пјҢеӣ жӯӨеҸҜд»Ҙи®ҫзҪ® $p[n + 1]=1$гҖӮ

    • жһҡдёҫеҹҺеёӮ $2\sim N$ зҡ„е…ЁжҺ’еҲ— $p$гҖӮ
      • еҸӘйңҖиҰҒ next_permutation(p + 2, p + n + 1)гҖӮжіЁж„ҸеҹҺеёӮ $1$ дёҚеҸӮдёҺе…ЁжҺ’еҲ—гҖӮ
    • еҜ№жҜҸдёӘжҺ’еҲ— $p$ и®Ўз®—пјҡ
    • еҲқе§ӢеҢ– $sum=0$ з”ЁжқҘи®Ўз®—з§»еҠЁжҖ»и·қзҰ»гҖӮ

    • зҙҜеҠ  $T[p[i]][p[i +1]]$ з§»еҠЁи·қзҰ»гҖӮ

    • иӢҘ $sum = K$пјҢеҲҷ ans++гҖӮ

    дҫӢйўҳеӣӣпјҡзјҶиҪҰ¶

    йўҳж„Ҹпјҡ$n$ еҗҚеӯҰз”ҹжҺ’йҳҹд№ҳзјҶиҪҰдёӢеұұпјҢзјҶиҪҰжңҖеӨ§жүҝйҮҚ $W$пјҢжҜҸиҫҶиҪҰиҙ№з”Ё $1$ е…ғгҖӮз¬¬ $i$ еҗҚеӯҰз”ҹдҪ“йҮҚ $a_i$гҖӮй—®жңҖе°‘йңҖиҰҒж”Ҝд»ҳеӨҡе°‘й’ұгҖӮ

    е®ҡд№ү $p[i]$ и®°еҪ•еӯҰз”ҹзј–еҸ·пјҢеҲқе§ӢеҢ– $p[i] = i$гҖӮ

    • жһҡдёҫеӯҰз”ҹзҡ„жүҖжңүеҸҜиғҪйЎәеәҸпјҲе…ЁжҺ’еҲ—пјүгҖӮ
    • еҜ№жҜҸдёҖз§ҚйЎәеәҸпјҢжҢүйЎәеәҸдҫқж¬Ўе°қиҜ•дёҠиҪҰпјҡ
      • з”ЁеҸҳйҮҸ $\texttt{sum}$ зҙҜи®ЎеҪ“еүҚзјҶиҪҰзҡ„иҪҪйҮҚгҖӮ
      • еҲқе§ӢеҢ– $\texttt{sum} = a[\texttt{p}[1]]$пјҢе·Із§ҹиҪҰиҫҶж•° $\texttt{cnt}=1$гҖӮ
      • еҜ№дҪҷдёӢжҜҸдҪҚеӯҰз”ҹ $\texttt{p}[i]$пјҡ
        • еҰӮжһң $\texttt{sum} + a[\texttt{p}[i]] \le W$пјҢеҲҷеҠ еҲ°еҪ“еүҚзјҶиҪҰпјҡ$\texttt{sum} \mathrel{+}= a[\texttt{p}[i]]$гҖӮ
        • еҗҰеҲҷеҸҰз§ҹдёҖиҫҶпјҡ$\texttt{cnt} \mathrel{+}= 1,\;\texttt{sum} = a[\texttt{p}[i]]$.
    • и®°еҪ•жүҖжңүйЎәеәҸдёӯжңҖе°Ҹзҡ„ $\texttt{cnt}$гҖӮ
    • ж—¶й—ҙеӨҚжқӮеәҰпјҡжһҡдёҫ $n!$ з§ҚйЎәеәҸпјҢжҜҸз§ҚйЎәеәҸеҒҡ $O(n)$ иЈ…иҪҰж“ҚдҪңпјҢйҖӮеҗҲ $n\le10$гҖӮ

    и®ҫ $n=3$, $W=100$пјҢдҪ“йҮҚж•°з»„ ${30, 80, 50}$гҖӮ
    жһҡдёҫйЎәеәҸ $(30,50,80)$пјҡ

    • з¬¬дёҖдҪҚ $30$пјҡж–°иҪҰпјҢ$\texttt{sum}=30$пјҢ$\texttt{cnt}=1$гҖӮ
    • з¬¬дәҢдҪҚ $50$пјҡ$30+50\le100$пјҢеҗҢиҪҰпјҢ$\texttt{sum}=80$гҖӮ
    • з¬¬дёүдҪҚ $80$пјҡ$80+80>100$пјҢж–°иҪҰпјҢ$\texttt{cnt}=2$пјҢ$\texttt{sum}=80$гҖӮ
    • жңҖз»ҲйңҖиҰҒ $2$ иҫҶзјҶиҪҰгҖӮ

    еҜ№жүҖжңүжҺ’еҲ—йғҪеҸҜеҒҡеҗҢж ·и®Ўз®—пјҢеҸ–жңҖе°Ҹ $\texttt{cnt}$ еҚідёәзӯ”жЎҲгҖӮ

    дҫӢйўҳдә”пјҡ[ABC363C] Avoid K Palindrome 2¶

    йўҳзӣ®йңҖиҰҒжҺ’еҲ—еӯ—з¬ҰдёІ $s$пјҢдё”з”ұдәҺ $|s|\leq 10$пјҢеӣ жӯӨиҖғиҷ‘дҪҝз”Ёе…ЁжҺ’еҲ—жқҘеҒҡгҖӮ

    жіЁж„Ҹжң¬йўҳж ·дҫӢ $2$ зҡ„еӯ—з¬ҰдёІ zzyyxпјҢиҝҷдёӘеӯ—з¬ҰдёІдёҖе…ұеҸӘжңү $30$ дёӘжҺ’еҲ—иҖҢдёҚжҳҜ $5!$ дёӘгҖӮеҺҹеӣ еңЁдәҺиҝҷйҮҢжңүдёӨдёӘйҮҚеӨҚзҡ„еӯ—з¬Ұ $z,y$гҖӮеӣ жӯӨжӯЈзЎ®зҡ„е…ЁжҺ’еҲ—дёӘж•°дёә $\dfrac{5!}{2!\times 2!}$гҖӮ

    еҰӮжһңиҝҳжҳҜжІҝз”Ёд№ӢеүҚзҡ„еҒҡжі•пјҲе…ҲеҲқе§ӢеҢ–йЎәеәҸж•°з»„ $a$пјүеҲҷдјҡз”ҹжҲҗ $120$ дёӘжҺ’еҲ—пјҢдјҡеҜјиҮҙжҹҗдәӣйҮҚеӨҚзҡ„жҺ’еҲ—иў«и®Ўз®—еӨҡж¬ЎпјҢд»ҺиҖҢдҪҝеҫ—зӯ”жЎҲеўһеӨҡгҖӮ

    еӣ жӯӨжң¬йўҳиҰҒеҜ№еӯ—з¬ҰдёІжң¬иә«иҝӣиЎҢе…ЁжҺ’еҲ—пјҢе®ғзҡ„з”Ёжі•зұ»дјјдәҺеҜ№еӯ—з¬ҰдёІзӣҙжҺҘжҺ’еәҸгҖӮ

    еҚі next_permutation(s.begin(), s.end())гҖӮ

    жіЁж„ҸдҪҝз”Ёд№ӢеүҚйңҖиҰҒе…ҲдҝқиҜҒ $s$ еҲқе§ӢжҳҜеӯ—е…ёеәҸжңҖе°Ҹзҡ„жғ…еҶөпјҢеҚіи°ғз”Ё sort(s.begin(), s.end()) е…ҲеҲқе§ӢеҢ–дёәеӯ—е…ёеәҸжңҖе°Ҹзҡ„жғ…еҶөгҖӮ

    жҺҘдёӢжқҘеҚіеҜ№дәҺдёҖдёӘеӯ—з¬ҰдёІ $s$пјҢйӘҢиҜҒжҳҜеҗҰе…·жңүй•ҝеәҰдёә $k$ еҫ—еӣһж–ҮеӯҗдёІпјҢеҸҜд»Ҙз”Ёжү“ж Үи®°еҫ—жҖқжғіе®һзҺ°гҖӮ

    • е®ҡд№ү bool check(l, r) зҡ„еҮҪж•°з”ЁдәҺеҲӨж–ӯеӯ—з¬ҰдёІ $s$ зҡ„еҢәй—ҙ $$l\sim r$ жҳҜеҗҰеӣһж–ҮгҖӮ
    • жҺҘдёӢжқҘе°ұжҳҜжһҡдёҫжүҖжңүй•ҝеәҰдёә $k$ зҡ„еҢәй—ҙпјҢз”ұдәҺеҢәй—ҙй•ҝеәҰеӣәе®ҡдё”жҳҜеӯҗдёІпјҢжһҡдёҫе·Ұз«ҜзӮ№ $i$ еҚіеҸҜпјҢеҚіеҸҜеҗҢжӯҘи®Ўз®—еҮәеҸіз«ҜзӮ№гҖӮ
    • иӢҘеӯҳеңЁдёҖдёӘеҢәй—ҙжҳҜеӣһж–Үзҡ„пјҢеҲҷеҪ“еүҚжҺ’еҲ—жҳҜдёҚеҗҲжі•зҡ„пјҢжү“дёҠж Үи®°гҖӮ
    • жңҖеҗҺж №жҚ®ж Үи®°еҸҳйҮҸзҡ„еҖјжқҘеҲӨе®ҡиҝҷдёӘеӯ—з¬ҰдёІжҳҜеҗҰз¬ҰеҗҲиҰҒжұӮеҚіеҸҜгҖӮ

    ж—¶й—ҙеӨҚжқӮеәҰпјҡ$O(n!\times n^2)$

    дҫӢйўҳе…ӯпјҡжңҖе°‘ж“ҚдҪңж•°¶

    дәӨжҚўзӣёйӮ»дёӨиЎҢжҲ–еҲ—еҸҜд»Ҙзұ»жҜ”еҒҡеҶ’жіЎжҺ’еәҸпјҢеӣ жӯӨиЎҢжҲ–еҲ—зҡ„дәӨжҚўе®һйҷ…е°ұжҳҜеңЁеҜ№иЎҢжҲ–еҲ—йҮҚж–°жҺ’еҲ—гҖӮ

    еӣ жӯӨжң¬йўҳйңҖиҰҒе…ЁжҺ’еҲ—жүҖжңүзҡ„еҸҜиғҪпјҢдёҖе…ұ $n!\times m!$ з§ҚпјҲеөҢеҘ—зҡ„е…ізі»пјүгҖӮ

    еӣ жӯӨйңҖиҰҒдёӨдёӘзј–еҸ·ж•°з»„ $a$ е’Ң $b$ еҲҶеҲ«и®°еҪ•иЎҢе’ҢеҲ—зҡ„зј–еҸ·пјҢеҲқе§ӢеҢ–еҲҶеҲ«дёә $1\sim n$ дёҺ $1\sim m$гҖӮ

    еҰӮдҪ•и®Ўз®—дәӨжҚўж¬Ўж•°пјҹ

    • еҜ№дәҺз»ҷе®ҡзҡ„жҺ’еҲ— $a$пјҢе®ғеҲқе§ӢжҳҜ $1\sim n$пјҢеҸҳеҢ–дёәе…¶е®ғжҺ’еҲ—еҗҺпјҢеҸӘйңҖиҰҒи®Ўз®—еҮәиҝҷдёӘжҺ’еҲ—зҡ„йҖҶеәҸеҜ№дёӘж•°еҚіеҸҜпјҢйҖҶеәҸеҜ№дёӘж•°е°ұжҳҜе®ғзҡ„дәӨжҚўж¬Ўж•°гҖӮ
      • йҖҶеәҸеҜ№пјҡеҪ“ж»Ўи¶і $ia_j$ е°ұз§° $(i,j)$ жҳҜдёҖдёӘйҖҶеәҸеҜ№гҖӮ
      • еӣ дёәеҶ’жіЎжҺ’еәҸжң¬иҙЁе°ұжҳҜж¶ҲйҷӨйҖҶеәҸеҜ№зҡ„иҝҮзЁӢгҖӮ
    • жҖ»дәӨжҚўж¬Ўж•°е°ұзӯүдәҺиЎҢзҡ„дәӨжҚўж¬Ўж•°еҠ дёҠеҲ—зҡ„дәӨжҚўж¬Ўж•°гҖӮжіЁж„ҸжҳҜеҠ дёҚжҳҜд№ҳгҖӮ
    • ж—¶й—ҙеӨҚжқӮеәҰдёәпјҡ$O(n!\times m!\times n^2\times m^2)$гҖӮ

    дҫӢйўҳдёғпјҡ[ABC150C] Count Order¶

    з®ҖеҚ•йўҳпјҢеҜ№жҺ’еҲ— $1\sim n$ иҝӣиЎҢе…ЁжҺ’еҲ—пјҢеҗҢж—¶и®ҫзҪ®дёҖдёӘеҸҳйҮҸи®°еҪ•зӣ®еүҚеҲ°дәҶз¬¬еҮ дёӘжҺ’еҲ—пјҢз„¶еҗҺеҲҶеҲ«жЈҖйӘҢеҪ“еүҚжҺ’еҲ—жҳҜеҗҰеҲҶеҲ«зӯүдәҺ $p$ жҲ– $q$гҖӮ

    дҫӢйўҳе…«пјҡ[ABC198D] Send More Money¶

    йўҳж„Ҹпјҡз»ҷдҪ дёүдёӘеӯ—з¬ҰдёІ $a,b,c$пјҢе°ҶжҜҸдёӘеӯ—з¬ҰзңӢдҪңжҳҜдёҖдёӘ $0\sim 9$ д№Ӣй—ҙзҡ„ж•°еӯ—пјҢеҸҜд»Ҙеҫ—еҲ°ж–°зҡ„дёүдёӘж•°еӯ— $N_1,N_2,N_3$гҖӮиҰҒжұӮпјҡ

    • жҜҸдёӘеӯ—з¬ҰеҜ№еә”зҡ„ж•°еӯ—е”ҜдёҖпјҢеҚіиӢҘеӯ—жҜҚ $a$ зңӢдҪңжҳҜ 1пјҢе…¶д»–еӯ—жҜҚеҲҷдёҚиғҪз”Ё $1$гҖӮ
    • ж–°зҡ„дёүдёӘж•°еӯ—ж— еүҚеҜј $0$ дё”йғҪжҳҜжӯЈж•ҙж•°гҖӮ
    • ж»Ўи¶ід»ҘдёҠжқЎд»¶еҗҺз»ҷе®ҡдёҖдёӘжһ„йҖ ж–№жЎҲгҖӮ

    йҰ–е…ҲеҸҜд»Ҙз»ҹи®ЎдёүдёӘеӯ—з¬ҰдёІдёӯдёҖе…ұеҮәзҺ°еӨҡе°‘дёӘдёҚеҗҢзҡ„еӯ—з¬ҰпјҢиӢҘдёҚеҗҢзҡ„дёӘж•°еӨ§дәҺ $10$ еҲҷж— и§ЈгҖӮ

    жҺҘдёӢжқҘеҸҜд»ҘеҜ№жҜҸдёҖдёӘеӯ—з¬ҰйҮҚж–°жҳ е°„жҲҗдёҖдёӘж•°еӯ—пјҢдҫӢеҰӮжҠҠ $a$ зңӢдҪң $1$пјҢд»ҘжӯӨзұ»жҺЁгҖӮжіЁж„ҸиҝҷйҮҢеҮәзҺ°зҡ„еӯ—жҜҚдёҚдёҖе®ҡжҳҜжҢүз…§ $a,b,c$ иҝҷж ·иҝһз»ӯзҡ„пјҢеҸҜиғҪжҳҜе°Ҷ $a,d,z$ йҮҚж–°зј–еҸ·дёә $1,2,3$гҖӮе…·дҪ“еҸҜд»Ҙз”Ё map е®һзҺ°гҖӮ

    string a, b, c;
cin >> a >> b >> c;
string s = a + b + c; // жӢјеҲ°дёҖиө·
for (auto x : s) vis[x - 'a'] = 1;
int num = 0;
for (int i = 0; i < 26; i++) num += (vis[i] == 1);
if (num > 10)
{
    cout << "UNSOLVABLE";
    return 0;
}
map<char, int> id;
int cnt = 0;
for (int i = 0; i < 26; i++)
{
    if (vis[i]) id[i + 'a'] = ++cnt; // жҜҸдёӘеӯ—жҜҚйҮҚж–°з»ҷдёӘж•°еҖј
}

    жҺҘдёӢжқҘжһҡдёҫ $0\sim 9$ зҡ„жҺ’еҲ—пјҢжқҘйҮҚж–°жЈҖйӘҢжҳҜеҗҰз”ҹжҲҗзҡ„ж•°еҖјз¬ҰеҗҲиҰҒжұӮгҖӮ

    for (int i = 0; i < 10; i++) x[i] = i;
do
{
    if (!x[id[a[0]]] || !x[id[b[0]]]) continue;
    long long suma = 0;
    for (auto v : a) suma = suma * 10 + (x[id[v]]);
    long long sumb = 0;
    for (auto v : b) sumb = sumb * 10 + (x[id[v]]);
    long long sumc = 0;
    for (auto v : c) sumc = sumc * 10 + (x[id[v]]);
    if (suma && sumb && suma + sumb == sumc)

        return 0;
    }
