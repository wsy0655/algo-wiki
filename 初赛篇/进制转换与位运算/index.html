    еҗ«д№үпјҡиЎЁиҫҫејҸ x & -x иҝ”еӣһ $x$ дёӯжңҖдҪҺдҪҚзҡ„ $1$ еҜ№еә”зҡ„ еҖјпјҲдҫӢеҰӮ $4$гҖҒ$8$ зӯүпјүгҖӮ

    еҺҹзҗҶпјҲзӣҙи§ӮпјүпјҡеңЁдәҢиЎҘз ҒдёӢ $-x = \overline{x}+1$пјҢдёҺ $x$ жҢүдҪҚдёҺеҸӘдҝқз•ҷжңҖдҪҺзҡ„ 1гҖӮ

    жіЁж„Ҹиҙҹж•°иҰҒз®—еҮәиЎҘз ҒеҗҺеңЁи®Ўз®—гҖӮзҶҹзҹҘжң¬жҠҖе·§еҗ«д»ҘеҗҺе°ұеҸҜд»ҘзӣҙжҺҘеҲ©з”Ё $x$ зҡ„дәҢиҝӣеҲ¶жұӮеҫ—з»“жһңгҖӮ

    зӨәдҫӢпјҡ$x=12=1100_2$пјҢеҲҷ

    \[ x\&(-x) = 1100_2 \& 0100_2 = 0100_2 = 4. \]

    жё…йҷӨжңҖдҪҺдҪҚзҡ„ 1 вҖ”вҖ” x & (x - 1)

    еҗ«д№үпјҡиЎЁиҫҫејҸ x & (x - 1) е°ҶжҠҠ $x$ зҡ„жңҖдҪҺдҪҚзҡ„ $1$ еҸҳдёә $0$пјҢе…¶дҪҷдҪҚдёҚеҸҳгҖӮеёёз”ЁдәҺи®Ўж•°дәҢиҝӣеҲ¶дёӢ $1$ зҡ„дёӘж•°гҖӮ

    дҫӢеӯҗпјҡ$x=12=1100_2$пјҢ

    \[ x\&(x-1) = 1100_2 \& 1011_2 = 1000_2 = 8. \]

    еә”з”Ёпјҡи®Ўз®— $x$ дәҢиҝӣеҲ¶дёӢ $1$ зҡ„дёӘж•°зҡ„дёҖз§ҚеҶҷжі•пјҢж—¶й—ҙеӨҚжқӮеәҰд№ҹжҳҜ $O(\log{x})$гҖӮ

    int count(int x) // жұӮдәҢиҝӣеҲ¶дёӢжңүеҮ дёӘ 1
{
    int cnt = 0;
    while (x) 
    {
        x &= x - 1;
        ++cnt;
    }
    return cnt;
}

    еҲӨж–ӯ $x$ жҳҜеҗҰдёә $2$ зҡ„е№Ӯ

    е…¬ејҸпјҡ

    \[ \text{isPow2}(x)\iff x>0\ \text{дё”}\ (x\& (x-1))==0. \]

    пјҲеӣ дёә $2$ зҡ„е№ӮеңЁдәҢиҝӣеҲ¶дёӯеҸӘжңүдёҖдёӘ $1$гҖӮпјү

    з»ғд№ 

    • йўҳпјҡ$x=16$пјҢеҲӨж–ӯжҳҜеҗҰдёә 2 зҡ„е№Ӯпјҹ
    • и§Јпјҡ$16\&(16-1)=16\&15=0$пјҢдё” $x>0$пјҢжүҖд»ҘжҳҜгҖӮ
    • зӯ”пјҡжҳҜпјҲ$\text{True}$пјүгҖӮвң…

    з»ҹи®Ў 1 зҡ„дёӘж•°пјҲpopcountпјү

    ж–№жі•пјҡдҪҝз”ЁеҶ…зҪ®еҮҪж•°пјҲеҰӮ GNU C++ зҡ„ __builtin_popcount(x)пјүжҲ–еҫӘзҺҜ x &= x - 1 жҲ–зӣҙжҺҘж•°дҪҚжӢҶеҲҶгҖӮ

    int count(int x)
{
    int cnt = 0;
    while (x)
    {
        if (x % 2 == 1) cnt++;
        x /= 2;
    }
    return cnt;
}

    еҹәжң¬еӯҳеӮЁеҚ•дҪҚ

    • еӯ—иҠӮпјҡbyteгҖӮ1 byte = 8 bit

    еӯ—иҠӮжҳҜи®Ўз®—жңәеҶ…еӯҳзҡ„еҹәжң¬еҚ•дҪҚпјҢиҖҢдҪҚжҳҜи®Ўз®—жңәиҝҗз®—зҡ„жңҖе°ҸеҚ•дҪҚгҖӮ
